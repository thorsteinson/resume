\newcommand{\proficiency}[1]{
  % Reset our counter
  \setcounter{ct}{0}
  \setcounter{proficiency}{#1}

  \forloop{ct}{0}{\value{ct} < 5}{
    % Decrement counter every time
    \addtocounter{proficiency}{-1}
    \ifthenelse{\value{proficiency} < 0}{\emptylevel}{\level}
  }
}
